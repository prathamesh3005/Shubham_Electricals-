 // Function to handle slider navigation
            function navigateSlider(direction) {
                const sliderContainer = document.querySelector('.slider-container');
                const slideWidth = document.querySelector('.slider').offsetWidth;
                const currentPosition = parseInt(sliderContainer.style.transform.slice(11)) || 0;
                const numSlides = document.querySelectorAll('.slider').length;
                const maxWidth = slideWidth * (numSlides - 1);

                let newPosition;

                if (direction === 'prev') {
                    newPosition = Math.min(currentPosition + slideWidth, 0);
                } else if (direction === 'next') {
                    newPosition = Math.max(currentPosition - slideWidth, -maxWidth);
                }

                sliderContainer.style.transform = `translateX(${newPosition}px)`;
            }

            // Automatic sliding
            function startSlider() {
                setInterval(() => {
                    navigateSlider('next');
                }, 10000); // Adjust the duration (in milliseconds) between each slide transition



                // Reset the slider to the beginning when it reaches the last slide
                const numSlides = document.querySelectorAll('.slider').length;
